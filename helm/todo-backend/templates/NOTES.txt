1. Backend is accessible within cluster at:
   http://{{ include "todo-backend.fullname" . }}:{{ .Values.service.port }}

2. Verify the deployment:
   kubectl get pods -l app=todo-backend -n {{ .Release.Namespace }}
   kubectl get svc {{ include "todo-backend.fullname" . }} -n {{ .Release.Namespace }}
   kubectl logs -l app=todo-backend -n {{ .Release.Namespace }} --tail=50

3. Check health:
   kubectl exec -it $(kubectl get pod -l app=todo-backend -n {{ .Release.Namespace }} -o jsonpath='{.items[0].metadata.name}') -n {{ .Release.Namespace }} -- \
     curl http://localhost:8000/api/health

4. Port forward for local testing:
   kubectl port-forward svc/{{ include "todo-backend.fullname" . }} 8000:8000 -n {{ .Release.Namespace }}
